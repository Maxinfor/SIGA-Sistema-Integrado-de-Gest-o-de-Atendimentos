/* ========================================================== SIGACTPAR - MÓDULO DE CRIANÇAS E ADOLESCENTES ========================================================== */ let criancaEditando = null; let criancaExcluirId = null; const fecharVisRodape = document.getElementById("fecharVisualizarCriancaRodape"); if (fecharVis) { fecharVis.onclick = () => { const modal = document.getElementById("modalVisualizarCrianca"); if (modal) modal.classList.remove("ativo"); const modal = document.getElementById("modalExcluirCrianca"); if (modal) modal.classList.remove("ativo"); criancaExcluirId = null; } }; } const pesquisa = document.getElementById("pesquisaCrianca"); if (pesquisa) pesquisa.onkeyup = atualizarTabelaCriancas; const form = document.getElementById("formCrianca"); if (form) form.onsubmit = salvarCrianca; } function atualizarIndicadoresCriancas() { if (!Banco || !Banco.dados || !Banco.dados.criancas) return; const lista = Banco.dados.criancas; const total = lista.length; const comEscola = lista.filter(c => c.escola && c.escola.trim() !== "").length; const cardTotal = document.getElementById("cardTotalCriancas"); const cardEscola = document.getElementById("cardComEscola"); if (cardTotal) cardTotal.textContent = total; if (cardEscola) cardEscola.textContent = comEscola; } function salvarCrianca(e) { e.preventDefault(); const objeto = { id: criancaEditando ?? gerarId("criancas"), nome: document.getElementById("nomeCrianca").value.trim(), nascimento: document.getElementById("nascimentoCrianca").value, cpf: document.getElementById("cpfCrianca").value.trim(), escola: document.getElementById("escolaCrianca").value.trim(), responsavel: document.getElementById("responsavelCrianca").value.trim(), observacoes: document.getElementById("observacoesCrianca").value.trim() }; if (criancaEditando === null) { inserirRegistro("criancas", objeto); } else { atualizarRegistro("criancas", objeto); } const modal = document.getElementById("modalCrianca"); if (modal) modal.classList.remove("ativo"); criancaEditando = null; atualizarTabelaCriancas(); atualizarIndicadoresCriancas(); } function visualizarCrianca(id) { if (!Banco || !Banco.dados || !Banco.dados.criancas) return; const item = Banco.dados.criancas.find(c => Number(c.id) === Number(id)); if (elObs) elObs.value = item.observacoes || ""; const modal = document.getElementById("modalCrianca"); if (modal) modal.classList.add("ativo"); } function excluirCrianca(id) { criancaExcluirId = id; const modal = document.getElementById("modalExcluirCrianca"); if (modal) modal.classList.add("ativo"); } function atualizarTabelaCriancas() { const tbody = document.getElementById("listaCriancas"); if (!tbody || !Banco || !Banco.dados || !Banco.dados.criancas) return; tbody.innerHTML = ""; const termo = document.getElementById("pesquisaCrianca")?.value.toLowerCase() || ""; let lista = Banco.dados.criancas.filter(item => (item.nome && item.nome.toLowerCase().includes(termo)) || (item.cpf && item.cpf.toLowerCase().includes(termo)) || (item.escola && item.escola.toLowerCase().includes(termo)) ); if (lista.length === 0) { tbody.innerHTML = `Nenhuma criança ou adolescente cadastrado.`; return; } tbody.innerHTML = lista.map(item => ` #${item.id} ${item.nome} ${formatarData(item.nascimento)} ${item.escola} ${item.responsavel}
`).join(""); } function formatarData(dataStr) { if (!dataStr) return ""; const partes = dataStr.split("-"); if (partes.length !== 3) return dataStr; return `${partes[2]}/${partes[1]}/${partes[0]}`; }
