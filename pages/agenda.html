/* ========================================================== SIGACTPAR - MÓDULO DE AGENDA E COMPROMISSOS ========================================================== */ let agendaEditando = null; let agendaExcluirId = null; document.addEventListener("DOMContentLoaded", () => { if (typeof iniciarAgenda === "function") { iniciarAgenda(); } }); function iniciarAgenda() { configurarEventosAgenda(); atualizarTabelaAgenda(); atualizarIndicadoresAgenda(); } function configurarEventosAgenda() { const btnNovo = document.getElementById("btnNovoCompromisso"); if (btnNovo) btnNovo.onclick = abrirModalAgendaNova; const btnAtualizar = document.getElementById("btnAtualizarAgenda"); if (btnAtualizar) btnAtualizar.onclick = () => { atualizarTabelaAgenda(); atualizarIndicadoresAgenda(); }; const fecharModal = document.getElementById("fecharModalAgenda"); const cancelarModal = document.getElementById("cancelarAgenda"); if (fecharModal) fecharModal.onclick = fecharModalAgenda; if (cancelarModal) cancelarModal.onclick = fecharModalAgenda; const form = document.getElementById("formAgenda"); if (form) form.onsubmit = salvarCompromisso; const pesquisa = document.getElementById("pesquisaAgenda"); if (pesquisa) pesquisa.onkeyup = atualizarTabelaAgenda; } function abrirModalAgendaNova() { agendaEditando = null; const form = document.getElementById("formAgenda"); if (form) form.reset(); const modal = document.getElementById("modalAgenda"); if (modal) modal.classList.add("ativo"); } function fecharModalAgenda() { const modal = document.getElementById("modalAgenda"); if (modal) modal.classList.remove("ativo"); agendaEditando = null; } function salvarCompromisso(e) { e.preventDefault(); const titulo = document.getElementById("tituloAgenda").value.trim(); const dataHora = document.getElementById("dataAgenda").value; const local = document.getElementById("localAgenda").value.trim(); const responsavel = document.getElementById("responsavelAgenda").value.trim(); if (!titulo || !dataHora || !local) { alert("Preencha todos os campos obrigatórios (*)."); return; } const objetoAgenda = { id: agendaEditando ?? (typeof gerarId === "function" ? gerarId("agenda") : Date.now()), titulo, dataHora, local, responsavel: responsavel || "Geral" }; if (agendaEditando === null) { if (typeof inserirRegistro === "function") { inserirRegistro("agenda", objetoAgenda); } else { if (!Banco.dados.agenda) Banco.dados.agenda = []; Banco.dados.agenda.push(objetoAgenda); if (typeof salvarBanco === "function") salvarBanco(); } } else { if (typeof atualizarRegistro === "function") { atualizarRegistro("agenda", objetoAgenda); } else { let index = Banco.dados.agenda.findIndex(a => a.id === agendaEditando); if (index !== -1) { Banco.dados.agenda[index] = objetoAgenda; if (typeof salvarBanco === "function") salvarBanco(); } } } atualizarTabelaAgenda(); atualizarIndicadoresAgenda(); fecharModalAgenda(); } function atualizarTabelaAgenda() { const tbody = document.getElementById("listaAgenda"); if (!tbody) return; tbody.innerHTML = ""; const termoPesquisa = document.getElementById("pesquisaAgenda")?.value.toLowerCase() || ""; const lista = (Banco.dados && Banco.dados.agenda) ? Banco.dados.agenda : []; const filtradas = lista.filter(a => { const titulo = a.titulo ? a.titulo.toLowerCase() : ""; const local = a.local ? a.local.toLowerCase() : ""; return titulo.includes(termoPesquisa) || local.includes(termoPesquisa); }); filtradas.sort((a, b) => new Date(a.dataHora) - new Date(b.dataHora)); if (filtradas.length === 0) { tbody.innerHTML = `Nenhum compromisso encontrado.`; return; } tbody.innerHTML = filtradas.map(a => ` ${formatarDataHora(a.dataHora)} ${a.titulo} ${a.local} ${a.responsavel}
`).join(""); } function atualizarIndicadoresAgenda() { const lista = (Banco.dados && Banco.dados.agenda) ? Banco.dados.agenda : []; const total = lista.length; const hoje = new Date(); const proximos = lista.filter(a => new Date(a.dataHora) >= hoje).length; const elTotal = document.getElementById("cardTotalAgenda"); const elProximos = document.getElementById("cardAgendaProximos"); if (elTotal) elTotal.textContent = total; if (elProximos) elProximos.textContent = proximos; } function editarCompromisso(id) { const lista = (Banco.dados && Banco.dados.agenda) ? Banco.dados.agenda : []; const a = lista.find(item => Number(item.id) === Number(id)); if (!a) return; agendaEditando = a.id; document.getElementById("tituloAgenda").value = a.titulo || ""; document.getElementById("dataAgenda").value = a.dataHora || ""; document.getElementById("localAgenda").value = a.local || ""; document.getElementById("responsavelAgenda").value = a.responsavel || ""; const modal = document.getElementById("modalAgenda"); if (modal) modal.classList.add("ativo"); } function excluirCompromisso(id) { if (!confirm("Deseja realmente excluir este compromisso da agenda?")) return; if (typeof removerRegistro === "function") { removerRegistro("agenda", id); } else { Banco.dados.agenda = Banco.dados.agenda.filter(a => Number(a.id) !== Number(id)); if (typeof salvarBanco === "function") salvarBanco(); } atualizarTabelaAgenda(); atualizarIndicadoresAgenda(); } function formatarDataHora(dataHoraStr) { if (!dataHoraStr) return ""; const dataObj = new Date(dataHoraStr); if (isNaN(dataObj)) return dataHoraStr; return dataObj.toLocaleString("pt-BR", { day: "2-digit", month: "2-digit", year: "numeric", hour: "2-digit", minute: "2-digit" }); }
